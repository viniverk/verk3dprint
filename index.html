function gerarEsalvar() { // Entradas do usuário const cliente = document.getElementById('cliente').value; const peso = parseFloat(document.getElementById('peso').value); const tempo = parseFloat(document.getElementById('tempo').value); // Parâmetros do seu negócio (você pode deixar fixos ou criar inputs no HTML) const custoPorGrama = 0.10; // R$ 100 por kg / 1000g const valorHoraMaquina = 5.00; // Defina aqui o seu custo/hora (energia + desgaste) const margemLucro = 0.50; // 50% de margem sobre o custo total // Cálculos const custoMaterial = peso * custoPorGrama; const custoOperacao = tempo * valorHoraMaquina; const custoTotal = custoMaterial + custoOperacao; const valorFinal = custoTotal * (1 + margemLucro); // Salvar no LocalStorage (Histórico) const orcamento = { cliente, valor: valorFinal.toFixed(2), data: new Date().toLocaleDateString() }; const lista = JSON.parse(localStorage.getItem('orcamentos') || '[]'); lista.push(orcamento); localStorage.setItem('orcamentos', JSON.stringify(lista)); // Gerar PDF const { jsPDF } = window.jspdf; const doc = new jsPDF(); doc.setFontSize(18); doc.text("Orcamento: " + cliente, 10, 20); doc.setFontSize(14); doc.text("Valor Final: R$ " + valorFinal.toFixed(2), 10, 40); doc.save("orcamento_" + cliente + ".pdf"); exibirHistorico(); }
